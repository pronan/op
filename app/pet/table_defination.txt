CREATE TABLE `pet` (
  `id` int(10) unsigned NOT NULL AUTO_INCREMENT,
  `age` int(11) NOT NULL,
  `dad` int(10) unsigned NOT NULL,
  `name` varchar(50) NOT NULL,
  `mom` int(10) unsigned NOT NULL,
  PRIMARY KEY (`id`),
  UNIQUE KEY `id` (`id`),
  KEY `dad` (`dad`),
  KEY `mom` (`mom`),
  CONSTRAINT `pet_ibfk_1` FOREIGN KEY (`dad`) REFERENCES `user` (`id`),
  CONSTRAINT `pet_ibfk_2` FOREIGN KEY (`mom`) REFERENCES `user` (`id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8
